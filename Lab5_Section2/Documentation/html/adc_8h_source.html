542  /**
543  * ADC calibrate.
544  * Calibrates the ADC before first use.
545  *
546  * @note Set up the ADC clock and resolution before calibration
547  *
548  * @return E_NO_ERROR Calibration successful
549  * @return E_CALIBRATE_FAIL Failed calibration
550  */
551  static ErrorCode calibrate() {
552 
553  // Save current SC3 as modified
554  uint8_t sc3 = adc().SC3;
555 
556  // Start calibration
557  setAveraging(AdcAveraging_Cal);
558 
559  // Wait for calibration to complete
560  while ((adc().SC1[0] & ADC_SC1_COCO_MASK) == 0) {
561  __asm__("nop");
562  }
563 
564  // Clear COCO
565  (void)adc().R[0];
566 
567  // Check if calibration failed
568  bool failed = adc().SC3 & ADC_SC3_CALF_MASK;
569 
570  // Restore original SC3 value
571  adc().SC3 = sc3;
572 
573  // Check calibration outcome
574  if(failed) {
575  // Failed calibration
576  return setErrorCode(E_CALIBRATE_FAIL);
577  }
578 
579  // Calibration factor
580  uint16_t calib;
581  calib = adc().CLPS + adc().CLP4 + adc().CLP3 + adc().CLP2 + adc().CLP1 + adc().CLP0;
582  calib /= 2;
583  calib |= (1<<15); // Set MSB
584  adc().PG = calib;
585 
586 #ifdef ADC_MG_MG_MASK
587  calib = adc().CLMS + adc().CLM4 + adc().CLM3 + adc().CLM2 + adc().CLM1 + adc().CLM0;
588  calib /= 2;
589  calib |= (1<<15); // Set MSB
590  adc().MG = calib;
591 #endif
592 
593  return E_NO_ERROR;
594  }

603  static void enableComparison(AdcCompare adcCompare, int low=INT_MIN, int high=INT_MAX) {
604 #ifdef DEBUG_BUILD
605  usbdm_assert (low<=high, "ADC Low level > high level");
606 #endif
607  // Juggle CV1, CV2 values to satisfy comparison rules
608  switch (adcCompare) {
609  case AdcCompare_Disabled:
610  break;
611  case AdcCompare_LessThan:
612  case AdcCompare_GreaterThanOrEqual:
613  adc().CV1 = low;
614  break;
615  case AdcCompare_OutsideRangeExclusive:
616  case AdcCompare_InsideRangeInclusive:
617  adc().CV1 = low;
618  adc().CV2 = high;
619  break;
620  case AdcCompare_InsideRangeExclusive:
621  case AdcCompare_OutsideRangeInclusive:
622  adc().CV1 = high;
623  adc().CV2 = low;
624  break;
625  }
626  // Set comparison fields
627  adc().SC2 |= (adc().SC2&~(ADC_SC2_ACFE(1)|ADC_SC2_ACFGT(1)|ADC_SC2_ACREN(1)))|
628  (adcCompare&(ADC_SC2_ACFE(1)|ADC_SC2_ACFGT(1)|ADC_SC2_ACREN(1)));
629  }
