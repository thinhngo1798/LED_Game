852 /**
853  * Transmit and receive a series of values
854  *
855  * @tparam T Type for data transfer (may be inferred from parameters)
856  *
857  * @param[in] dataSize Number of values to transfer
858  * @param[in] txData Transmit bytes (may be nullptr for Receive only)
859  * @param[out] rxData Receive byte buffer (may be nullptr for Transmit only)
860  *
861  * @note: rxData may use same buffer as txData
862  * @note: Size of txData and rxData should be appropriate for transmission size.
863  */
864 template<typename T>
865 void __attribute__((noinline)) Spi::txRx(uint32_t dataSize, const T *txData, T *rxData) {
866 
867  static_assert (((sizeof(T) == 1)||(sizeof(T) == 2)), "Size of data type T must be 8 or 16-bits");
868 
869  while(dataSize-->0) {
870  uint32_t sendData = 0xFFFF;
871  if (txData != nullptr) {
872  sendData = (uint16_t)*txData++;
873  }
874  if (dataSize == 0) {
875  // Mark last data
876  sendData |= SPI_PUSHR_EOQ_MASK;
877  }
878  else {
879  // Keep SPI_PCS asserted between data values
880  sendData |= SPI_PUSHR_CONT_MASK;
881  }
882  spi->PUSHR = sendData|pushrMask;
883  while ((spi->SR & SPI_SR_TCF_MASK)==0) {
884  }
885  spi->SR = SPI_SR_TCF_MASK|SPI_SR_EOQF_MASK;
886  uint32_t receiveData = spi->POPR;
887  if (rxData != nullptr) {
888  *rxData++ = receiveData;
889  }
890  }
891  // Wait until tx/rx complete
892  while ((spi->SR&SPI_SR_TXRXS_MASK) == 0) {
893  __asm__("nop");
894  }
895 }
