145 /**
146  * Notes on the PDB.
147  *
148  * Components:\n
149  * - Channels+pre-triggers - Usually associated with an ADC
150  * - DAC delays - Associated with a DAC
151  * - Pulse outputs - Usually associated with a CMP
152  *
153  * Operation:\n
154  * - The PDB can be triggered from another peripheral or directly by software using softwareTrigger().
155  * - When triggered, the PDB counter and the DAC delay counters are restarted.
156  * - The period of the PDB counter and the delay counters may be independently controlled.
157  * - The PDB counter counts [0..MOD]. Set with setPeriod().
158  * - The DAC delay counters count [0..DACINTx]. Set with configureDacTrigger().
159  * The counters may be bypassed for external triggers.
160  * - The PDB may operate in one-shot or continuous mode.
161  * - One shot mode - The PDB and DAC delay counters are reset on trigger. The PDB counter counts one sequence only.
162  * The DAC counter only operates while the main counter is counting but it may produce multiple triggers in the sequence.
163  * - Continuous mode - As above but the counter resets when it reaches the counter period and restarts.\n
164  * The DAC delay counters are NOT reset on PDB counter roll-over so operate independently after initial trigger.
165  * - I would expect the DAC Delay counter period to be less than or equal to the PDB counter period in one-shot mode.
166  * - The DAC Delay period may be set smaller than (PDB counter period/2) to trigger multiple DAC output
167  * triggers within a PDB period but more often there would be only a single DAC event e.g.
168  * <b>(PDB counter period/2) < (DAC Delay period) <= (PDB counter period)</b>. Again this is in one-shot mode.
169  * - The pre-triggers within a channel are associated with different ADC pre-triggers e.g. SC1[n]/R[n]. The pre-triggers
170  * are referenced to the main PDB counter.\n
171  * This allows multiple ADC channels (inputs) to be used with a PDB period - usually only 2 are available per ADC.
172  * Set with configureAdcPretrigger().
173  * - The pulse outputs may be configured as high for a pulse from [start...end] times based on the PDB counter.
174  * Set with configurePulseOutput().
175  */

449  /**
450  * Get 'best' dividers for given period.
451  * This involves finding the smallest prescaler that allows the PDB period to be set to greater
452  * than the given period.\n
453  * This produces the highest resolution.\n
454  * It is quite possible that other values would be more suitable for a particular application.
455  * For example, carefully chosen prescaler may result in less rounding for the needed intermediate
456  * points for pulse outputs etc.
457  *
458  * @param[in] period Period in seconds as a float
459  * @param[out] multValue Determined pdb_sc_mult value
460  * @param[out] prescaleValue Determined pdb_sc_prescaler value
461  *
462  * @return E_NO_ERROR => success
463  * @return E_ERROR => failed to find suitable values
464  */
465  static ErrorCode getDividers(float period, uint32_t &multValue, int &prescaleValue) {
466 
467  // Multiplier factors for prescale divider
468  static const int multFactors[] = {1,10,20,40};
469 
470  float inputClock = Info::getInputClockFrequency();
471 
472  // No MOD value found so far
473  uint32_t mod = 0;
474 
475  // Try each divider multiplier
476  for (unsigned trialMultValue=0; trialMultValue<(sizeof(multFactors)/sizeof(multFactors[0])); trialMultValue++) {
477  int multfactor = multFactors[trialMultValue];
478 
479  // Try prescalers from smallest to largest
480  // Find first prescaler for which a suitable modulo exists
481  int prescaleFactor=1;
482  for (unsigned trialPrescaleValue=0; trialPrescaleValue<=7; trialPrescaleValue++) {
483  float clock = inputClock/(multfactor*prescaleFactor);
484  uint32_t trialMod = round(period*clock)-1;
485 // console.
486 // write("multfactor = ").write(multfactor).
487 // write(", prescaleFactor = ").write(prescaleFactor).
488 // write(", mod = ").write(trialMod).
489 // write(", period = ").writeln(period);
490  if (trialMod <= 0) {
491  // Too short a period
492  return E_TOO_SMALL;
493  }
494  if (trialMod <= 65535) {
495  if (trialMod>mod) {
496  // Better value - save
497  prescaleValue = trialPrescaleValue;
498  multValue = trialMultValue;
499  mod = trialMod;
500  }
501  break;
502  }
503  prescaleFactor <<= 1;
504  }
505  }
506  return setErrorCode((mod==0)?E_ERROR:E_NO_ERROR);
507  }

509  /**
510  * Sets period of main counter to given value.
511  *
512  * It attempts to get the 'best' dividers for a given period.\n
513  * This involves finding the smallest prescaler that allows the PDB period
514  * to be set to greater than the given period.\n
515  * This produces the highest resolution.\n
516  * It is quite possible that other values would be more suitable for a particular application.\n
517  * For example, carefully chosen prescalers may result in less rounding for the needed intermediate
518  * points for pulse outputs etc.
519  *
520  * @param[in] period Period in seconds as a float
521  *
522  * @return E_NO_ERROR => success
523  * @return E_ERROR => failed to find suitable values
524  *
525  * @note This affects pdb_sc_mult, pdb_sc_prescaler, pdb_mod
526  */
527  static ErrorCode setPeriod(float period) {
528 
529  uint32_t mult = 0;
530  int prescale = 0;
531 
532  ErrorCode rc = getDividers(period, mult, prescale);
533  if (rc != E_NO_ERROR) {

535  }
536  pdb().SC = (pdb().SC&~(PDB_SC_MULT_MASK|PDB_SC_PRESCALER_MASK))|
537  PDB_SC_MULT(mult)|PDB_SC_PRESCALER(prescale)|PDB_SC_PDBIF_MASK;
538 
539  // Calculate MOD using new MULT and PRESCALER convertSecondsToTicks()
540  pdb().MOD = convertSecondsToTicks(period) - 1;
541 
542  return E_NO_ERROR;
543  }
