301  /**
302  * Sets how often the inputs are scanned.
303  *
304  * @param[in] period The period of scanning.
305  *
306  * @note This period must be larger than the total measurement time as determined by other parameters e.g.
307  * - Electrode oscillation period (electrode/finger capacitance, electrode current)
308  * - Electrode oscillator divider and consecutive scans settings
309  * - Number of electrodes scanned.
310  */
311  static ErrorCode setSamplePeriod(float period) {
312  float inputClock = Info::getInputClockFrequency();
313  int prescaleFactor=1;
314  int prescalerValue=0;
315 
316  // Maximum period value in ticks
317  uint32_t maxPeriodInTicks = 255;
318 
319  while (prescalerValue<=7) {
320  float clock = inputClock/prescaleFactor;
321  uint32_t periodInTicks = round(period*clock);
322  if (periodInTicks <= maxPeriodInTicks) {
323  tsi().SCANC = (tsi().SCANC&~(TSI_SCANC_AMPSC_MASK|TSI_SCANC_SMOD_MASK))|
324  TSI_SCANC_AMPSC(prescalerValue)|TSI_SCANC_SMOD(periodInTicks);
325  return E_NO_ERROR;
326  }
327  prescalerValue++;
328  prescaleFactor <<= 1;
329  }
330  // Too long a period
331  return setErrorCode(E_TOO_LARGE);
332  }
