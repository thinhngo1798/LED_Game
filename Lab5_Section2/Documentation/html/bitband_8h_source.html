12 /*!
13  * @addtogroup BITBAND_group Bit-band access
14  * @brief Cortex-M4 bit-band support
15  * @{
16  *
17  * Based on Freescale Application Note AN4838
18  * http://cache.freescale.com/files/microcontrollers/doc/app_note/AN4838.pdf
19  *
20  * Assumes the location has been allocated to:
21  * - SRAM_U (0x2000_0000-0x200F_FFFF) with bit-band alias (0x2200_0000-0x23FF_FFFF)
22  * - AIPS/GPIO (0x4000_0000-0x400F_FFFF) with bit-band alias (0x4200_0000-0x43FF_FFFF)
23  *
24  * Most Global or Local Static variables end up in SRAM_L so are not usable\n
25  * Most Parameters and Local variables end up in SRAM_U
26  *
27  * Examples:
28  *
29  * Write a bit to a register
30  * @code{.c}
31  * bitbandWrite(GPIOA->PDOR, 3, true); // Set bit # 3 of port
32  * @endcode
33  *
34  * Set bit of register
35  * @code{.c}
36  * bitbandSet(SIM->SCGC5, SIM_SCGC5_PORTA_SHIFT); // Enable clock to PORTA
37  * @endcode
38  *
39  * Assuming a local variable 'local'
40  * @code{.c}
41  * bitbandClear(local, 3); // Clear bit 3 of local
42  * @endcode
43  *
44  * Note: The efficiency of this operation depends greatly on whether the address is a constant or can be pre-calculated.
45  * The code will usually be quite inefficient for local variables as the required address will be
46  * calculated each time i.e. the address calculation will overwhelm any advantage of the bitband operation
47  *
48  * It is probably a good idea to use a dedicated segment in SRAM_U for bit-band variables and access globally.
49  * See linker files.
50  *
51  * Fixed locations accessed by casts e.g. GPIOC etc will be very efficient since the calculation is a constant expression.
52  *
53  * The generated code will be very inefficient if optimization is not enabled.
54  */
55 
56 /**
57  * @brief Write a value to a bit.
58  * Only the referenced bit is modified.
59  *
60  * <b>Example</b> - Write value to bit of register
61  * @code{.c}
62  * bitbandWrite(GPIOA->PDOR, 3, true); // Set bit # 3 of port
63  * @endcode
64  *
65  * @param[out] ref Object to manipulate e.g. a register
66  * @param[in] bitNum Bit number
67  * @param[in] value Value to modify 0 or 1. Only the LSB is used
68  */
69 template <typename T>
70 static __attribute__((always_inline)) inline void bitbandWrite(T &ref, const int bitNum, uint32_t value) {
71  uint32_t addr = (uint32_t)(&ref);
72  const uint32_t mappedAddress = (addr&0xF0000000) + 0x02000000 + ((addr&0xFFFFF)*8*4) + (bitNum*4);
73  *(volatile uint32_t *)(mappedAddress) = value;
74 }

76 /**

80  * <b>Example</b> - Read value of bit withon a register

82  * if (bitbandRead(GPIOA->PDIR, 3) {
83  * // Only done if bit # 3 of port is high
84  * }
85  * @endcode
86  *
87  * @param[out] ref Object to examine e.g. a register
88  * @param[in] bitNum Bit number
89  *
90  * @return Bit read as boolean value
91  */
92 template <typename T>
93 static __attribute__((always_inline)) inline uint32_t bitbandRead(T &ref, const int bitNum) {
94  uint32_t addr = (uint32_t)(&ref);
95  const uint32_t mappedAddress = (addr&0xF0000000) + 0x02000000 + ((addr&0xFFFFF)*8*4) + (bitNum*4);
96  return *(volatile uint32_t *)(mappedAddress);
97 }
