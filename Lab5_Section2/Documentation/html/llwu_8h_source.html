328  /**

330  * The pin masks correspond to Pin sources.
331  *
332  * @code

342  if constexpr(PF_SIZE==4) {
343  return (llwu().PF[1]<<24)|(llwu().PF[1]<<16)|(llwu().PF[1]<<8)|llwu().PF[0];
344  }
345  else if constexpr(PF_SIZE==3) {
346  return (llwu().PF[1]<<16)|(llwu().PF[1]<<8)|llwu().PF[0];
347  }
348  else if constexpr(PF_SIZE==2) {
