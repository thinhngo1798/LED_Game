184  /**
185  * Set baud factor value for interface.
186  * For UARTS with baud rate fraction adjust (BRFA) support.
187  *
188  * This is calculated from baud rate and UART clock frequency
189  *
190  * @param[in] baudrate Interface speed in bits-per-second
191  * @param[in] clockFrequency Frequency of UART clock
192  */
193  void __attribute__((noinline)) setBaudRate_brfa(uint32_t baudrate, uint32_t clockFrequency) {
194 
195  /*
196  * Baudrate = clockFrequency / (OSR x (SBR + BRFD))
197  * Fixed OSR = 16
198  *
199  * (OSR x (SBR + BRFD/32)) = clockFrequency/Baudrate
200  * (SBR + BRFD/32) = clockFrequency/(Baudrate*OSR)
201  * 32*SBR + BRFD = 2*clockFrequency/Baudrate
202  * SBR = (2*clockFrequency/Baudrate)>>5
203  * BRFD = (2*clockFrequency/Baudrate)&0x1F
204  */
205  // Disable UART before changing registers
206  uint8_t c2Value = uart.C2;
207  uart.C2 = 0;
208 
209  // Calculate UART clock setting (5-bit fraction at right)
210  int divider = (2*clockFrequency)/baudrate;
211 
212  // Set Baud rate register
213  uart.BDH = (uart.BDH&~UART_BDH_SBR_MASK) | UART_BDH_SBR((divider>>(8+5)));
214  uart.BDL = UART_BDL_SBR(divider>>5);
215  // Fractional divider to get closer to the baud rate
216  uart.C4 = (uart.C4&~UART_C4_BRFA_MASK) | UART_C4_BRFA(divider);
217 
218  // Restore UART settings
219  uart.C2 = c2Value;
220  }

734 class UartBuffered_T : public Uart_T<Info> {
735 
736 public:
737  using Uart_T<Info>::uart;
738 
739  UartBuffered_T() : Uart_T<Info>() {
740  Uart::enableInterrupt(UartInterrupt_RxFull);
741  Uart_T<Info>::enableNvicInterrupts();
742  }
743 
744  virtual ~UartBuffered_T() {
745  Uart::enableInterrupt(UartInterrupt_RxFull, false);
746  Uart::enableInterrupt(UartInterrupt_TxHoldingEmpty, false);
747  }
748 
749  /**
750  * Queue for Buffered reception (if used)
751  */
752  static UartQueue<char, rxSize> rxQueue;
753  /**
754  * Queue for Buffered transmission (if used)
755  */
756  static UartQueue<char, txSize> txQueue;
757 
758 protected:
759 
760  /** Lock variable for writes */
761  static volatile uint32_t fWriteLock;
762 
763  /** Lock variable for reads */
764  static volatile uint32_t fReadLock;
765 
766  /**
767  * Writes a character (blocking on queue full)
768  *
769  * @param[in] ch - character to send
770  */
771  virtual void _writeChar(char ch) override {
772  lock(&fWriteLock);
773  // Add character to buffer
774  while (!txQueue.enQueueDiscardOnFull(ch)) {
775  }
776  uart.C2 |= UART_C2_TIE_MASK;
777  unlock(&fWriteLock);
778  if (ch=='\n') {
779  _writeChar('\r');
780  }
781  }
782 
783  /**
784  * Receives a single character (blocking on queue empty)
785  *
786  * @return Character received
787  */
788  virtual int _readChar() override {
789  lock(&fReadLock);
790  while (rxQueue.isEmpty()) {
791  __asm__("nop");
792  }
793  char t = rxQueue.deQueue();
794  unlock(&fReadLock);
795  return t;
796  }
797 
798  /**
799  * Check if character is available
800  *
801  * @return true Character available i.e. _readChar() will not block
802  * @return false No character available
803  */
804  virtual bool _isCharAvailable() override {
805  return (!rxQueue.isEmpty());
806  }
807 
808 public:
809  /**
810  * Receive/Transmit IRQ handler (MKL)
811  */
812  static void irqHandler() {
813  uint8_t status = Info::uart().S1;
814  if (status & UART_S1_RDRF_MASK) {
815  // Receive data register full - save data
816  rxQueue.enQueueDiscardOnFull(Info::uart().D);
817  }
818  if (status & UART_S1_TDRE_MASK) {
819  // Transmitter ready
820  if (txQueue.isEmpty()) {
821  // No data available - disable further transmit interrupts
822  Info::uart().C2 &= ~UART_C2_TIE_MASK;
823  }
824  else {
825  // Transmit next byte
826  Info::uart().D = txQueue.deQueue();
827  }
828  }
829  }
830 
831  /**
832  * Receive/Transmit IRQ handler (MK)
833  */
834  static void irqRxTxHandler() {
835  uint8_t status = Info::uart().S1;
836  if (status & UART_S1_RDRF_MASK) {
837  // Receive data register full - save data
838  rxQueue.enQueueDiscardOnFull(Info::uart().D);
839  }
840  if (status & UART_S1_TDRE_MASK) {
841  // Transmitter ready
842  if (txQueue.isEmpty()) {
843  // No data available - disable further transmit interrupts
844  Info::uart().C2 &= ~UART_C2_TIE_MASK;
845  }
846  else {
847  // Transmit next byte
848  Info::uart().D = txQueue.deQueue();
849  }
850  }
851  }
852 
853  /**
854  * Error IRQ handler (MK)
855  */
856  static void irqErrorHandler() {
857  // Ignore errors
858  clearError();
859  }
860 
861  /**
862  * Flush output data.
863  * This blocks until all pending data has been sent
864  */
865  virtual void flushOutput() override {
866  while (!txQueue.isEmpty()) {
867  // Wait until queue empty
868  }
869  while ((uart.S1 & UART_S1_TC_MASK) == 0) {
870  // Wait until transmission of last character is complete
871  }
872  }
873 
874  /**
875  * Flush input data
876  */
877  virtual void flushInput() override {
878  Uart_T<Info>::flushInput();
879  rxQueue.clear();
880  }
881 
882 };
