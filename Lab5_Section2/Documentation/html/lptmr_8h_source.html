432  /**
433  * Set period of timer.
434  *
435  * @param[in] period Period in seconds as a float
436  *
437  * @note Will enable and adjust the pre-scaler to appropriate value.\n
438  * The clock source should be selected by setClock() before using this function.
439  *
440  * @return E_NO_ERROR => Success
441  * @return E_ILLEGAL_PARAM => Failed to find suitable values for PBYP & PRESCALE
442  */
443  static ErrorCode setPeriod(float period) {
444  // Disable LPTMR before prescale change
445  uint32_t csr = lptmr().CSR;
446  lptmr().CSR = 0;
447 
448  float inputClock = Info::getInputClockFrequency();
449  int prescaleFactor=1;
450  uint32_t prescalerValue=0;
451  while (prescalerValue<=16) {
452  float clockFrequency = inputClock/prescaleFactor;
453  uint32_t mod = rintf(period*clockFrequency);
454  if (mod < MINIMUM_RESOLUTION) {
455  // Too short a period for reasonable resolution
456  return setAndCheckErrorCode(E_TOO_SMALL);
457  }
458  if (mod <= 65535) {
459  __DSB();
460  lptmr().CMR = mod;
461  lptmr().PSR = (lptmr().PSR & ~(LPTMR_PSR_PRESCALE_MASK|LPTMR_PSR_PBYP_MASK))|LPTMR_PSR_PRESCALE(prescalerValue-1)|LPTMR_PSR_PBYP(prescalerValue==0);
462  lptmr().CSR = csr;
463  return E_NO_ERROR;
464  }
465  prescalerValue++;
466  prescaleFactor <<= 1;
467  }
468  // Too long a period
469  return setAndCheckErrorCode(E_TOO_LARGE);
470  }

472  /**
473  * Set glitch filter interval.
474  * This adjusts the clock prescaler so that the filter interval is at least the given value.
475  *
476  * @param[in] interval Interval in seconds as a float
477  *
478  * @note Will enable and adjust the pre-scaler to appropriate value.\n
479  * The clock source should be selected by setClock() before using this function.
480  *
481  * @return E_NO_ERROR => Success
482  * @return E_ILLEGAL_PARAM => Failed to find suitable values for PBYP & PRESCALE
483  */
484  static ErrorCode setFilterInterval(float interval) {
485  long inputClock = Info::getInputClockFrequency();
486  int prescaleFactor=1;
487  uint32_t prescalerValue=0;
488  while (prescalerValue<=16) {
489  if ((interval*prescaleFactor) < inputClock) {
490  // Disable LPTMR before prescale change
491  uint32_t csr = lptmr().CSR;
492  lptmr().CSR = 0;
493  __DSB();
494  lptmr().PSR = (lptmr().PSR & ~(LPTMR_PSR_PRESCALE_MASK|LPTMR_PSR_PBYP_MASK))|LPTMR_PSR_PRESCALE(prescalerValue-1)|LPTMR_PSR_PBYP(prescalerValue==0);
495  lptmr().CSR = csr;
496  return E_NO_ERROR;
497  }
498  prescalerValue++;
499  prescaleFactor <<= 1;
500  }
501  // Too long a period
502  return setAndCheckErrorCode(E_TOO_LARGE);
503  }
