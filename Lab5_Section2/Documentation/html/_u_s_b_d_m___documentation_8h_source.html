160 Convenience template for GPIO pins. Uses the following classes:\n
161 <ul>
162 <li>USBDM::GpioBase_T &lt;clockMask, portAddress, gpioAddress, bitNum, polarity&gt; \n
163 <li>USBDM::Gpio_T &lt;Info, bitNum, polarity&gt;\n
164 <ul>
165 <li>USBDM::GpioA &lt;bitNum, polarity&gt;
166 <li>USBDM::GpioB &lt;bitNum, polarity&gt;
167 <li>USBDM::GpioC &lt;bitNum, polarity&gt;
168 <li>USBDM::GpioD &lt;bitNum, polarity&gt;
169 <li>USBDM::GpioE &lt;bitNum, polarity&gt;
170 </ul>
171 <li>USBDM::Field_T &lt;Info, left, right, polarity&gt;\n
172 <ul>
173 <li>USBDM::GpioAField &lt;left, right, polarity&gt;
174 <li>USBDM::GpioBField &lt;left, right, polarity&gt;
175 <li>USBDM::GpioCField &lt;left, right, polarity&gt;
176 <li>USBDM::GpioDField &lt;left, right, polarity&gt;
177 <li>USBDM::GpioEField &lt;left, right, polarity&gt;
178 </ul>
179 </ul>
180 
181 This template is an interface for the general purpose I/O pin hardware. \n
182 
183 It provides:\n
184 - Static pin mapping in conjunction with the configuration settings.
185 - Setting pin characteristics e.g. pull-ups, drive strength etc
186 - Changing the pin direction
187 - Setting outputs to high or low level
188 - Polling input value
189 - Polarity (\ref USBDM::ActiveHigh or \ref USBDM::ActiveLow) is selected when instantiated
190 
191 This is a template class with static methods.\n
192 <em>It cannot be instantiated.</em>
193

195  - @ref digital-example1.cpp
196  - @ref digital-example2.cpp

198 <b>Usage</b>
199  @code
200  // Open USBDM namespace.
201  // This allows access to USBDM classes and methods without the USBDM:: prefix.
202  using namespace USBDM;
203 
204  // Temporary for input values
205  bool x;
206 
207  // Use pin 3 of PORTC as active-high GPIO i.e. PTC3
208  using Gpio = GpioC<3, ActiveHigh>;
209 
210  // Set as digital output
211  Gpio::setOutput();
212  // or for more detailed control
213  Gpio::setOutput(PinDriveStrength_High, PinDriveMode_OpenDrain);
214 
215  // Set pin to active level (polarity is applied)
216  Gpio::setActive();
217  // or
218  Gpio::on();
219 
220  // Clear pin to inactive level (polarity is applied)
221  Gpio::setInactive();
222  // or
223  Gpio::off();
224 
225  // Set pin to a high level (polarity is ignored)
226  Gpio::set();
227  // or
228  Gpio::high();
229 
230  // Set pin to a low level (polarity is ignored)
231  Gpio::clear();
232  // or
233  Gpio::low();
234 
235  // Toggle pin
236  Gpio::toggle();
237 
238  // Set pin to boolean value (polarity is applied)
239  Gpio::write(true);
240 
241  // Set pin to boolean value (polarity is applied)
242  Gpio::write(false);
243 
244  // Enable pull-up
245  // This will only have effect when pin is an input
246  Gpio::setPullDevice(PinPull_Up);
247 
248  // Set drive strength
249  Gpio::setDriveStrength(PinDriveStrength_High);
250 
251  // Set open-drain
252  Gpio::setDriveMode(PinDriveMode_OpenDrain);
253 
254  // Read pin as boolean value (polarity is applied)
255  x = Gpio::read();
256 
257  // Read _state_ of pin drive as boolean value (polarity is applied)
258  // This may differ from the value on the pin
259  // May be useful if the pin is open-drain
260  Gpio::setDriveMode(PinOpenDrain);
261  Gpio::high();
262  if (Gpio::readState() != Gpio::read()) {
263  console.writeln("Open-drain pin is being held low");
264  }
265 
266  // Dynamically change pin to input
267  Gpio::setIn();
268 
269  // Read pin as boolean value (polarity is applied)
270  x = Gpio::read();
271 
272  // Read pin as boolean value (polarity is ignored)
273  x = Gpio::isHigh();
274 
275  // Read pin as boolean value (polarity is ignored)
276  x = Gpio::isLow();

363 @page FTMExamples Flexible Timer Module
364 
365 Convenience template for FTM hardware.
366 The interface is divided into a number of templates:
367 <ul>
368 <li>USBDM::FtmBase_T Representing the shared FTM functionality.
369 <ul>
370 <li>USBDM::Ftm0
371 <li>USBDM::Ftm1
372 </ul>
373 <li>USBDM::FtmQuadDecoder_T Representing a FTM operating as a quadrature encoder.
374 <ul>
375 <li>USBDM::FtmQuadDecoder1
376 </ul>
377 <li>USBDM::FtmBase_T::Channel Representing individual channels of a single FTM.
378 <ul>
379 <li>USBDM::Ftm0::Channel
380 <li>USBDM::Ftm1::Channel
381 </ul>
382 </ul>
383 It provides:\n
384 <ul>
385 <li>Static pin mapping in conjunction with the configuration settings.
386 <li>Setting the FTM period in ticks or seconds
387 <li>Setting the channel duty cycle in percentage
388 <li>Interrupt driven operation is also supported through a callback if <b>enabled in the configuration</b>.
389 </ul>
390 
391 This is a template class with static methods.\n
392 <em>It cannot be instantiated.</em>

401  <b>Usage - PWM</b>
402 @code
403  // Open USBDM namespace.
404  // This allows access to USBDM classes and methods without the USBDM:: prefix.
405  using namespace USBDM;
406 
407  // Initialise the timer with initial alignment
408  // This affects all channels of the FTM
409  Ftm0::configure(FtmMode_LeftAlign);
410 
411  // Set timer period in ticks
412  // This affects all channels of the FTM
413  // Timer pre-scaler is not changed
414  Ftm0::setPeriodInTicks(500);
415 
416  // Set timer period in microseconds
417  // This affects all channels of the FTM
418  // Timer pre-scaler (tick rate) will be adjusted to accommodate the required period
419  Ftm0::setPeriod(125*us);
420 
421  // Use FTM0 channel 3
422  using PwmOutput = Ftm0Channel<3> ;
423 
424  // Set channel to generate PWM with active-high pulses
425  PwmOutput::enable(ftm_pwmHighTruePulses);
426 
427  // Set duty cycle as percentage
428  PwmOutput::setDutyCycle(34);
429 
430  // Set duty cycle as percentage (float)
431  PwmOutput::setDutyCycle(12.25*percent);
432 
433  // Set high time in microseconds (float)
434  PwmOutput::setHighTime(63*us);
435 @endcode

530 <b>Usage - PDB Software Trigger</b>
531 @code
532  // Open USBDM namespace.
533  // This allows access to USBDM classes and methods without the USBDM:: prefix.
534  using namespace USBDM;
535 
536  Pdb::enable();
537  // Trigger from FTM
538  Pdb::setTriggerSource(PdbTrigger_Software);
539  // Set callback
540  Pdb::setCallback(pdbCallback);
541  // Interrupt during sequence
542  Pdb::enableSequenceInterrupts();
543  // Set period a bit longer than FTM period
544  Pdb::setPeriod(PERIOD);
545  // Generate interrupt near end of sequence
546  Pdb::setInterruptDelay(PERIOD-5*ms);
547  // Take ADC samples before and after sample edge
548  Pdb::setPretriggers(0, PdbPretrigger0_Delayed, HIGH_TIME-SAMPLE_DELAY, PdbPretrigger1_Delayed, HIGH_TIME+SAMPLE_DELAY);
549  // Update registers
550  Pdb::triggerRegisterLoad(PdbLoadMode_immediate);
551 
552  Pdb::enableNvicInterrupts();
553 
554  Pdb::softwareTrigger();
555 @endcode
