656  /**
657  * Set period.
658  *
659  * @param[in] period Period in ticks (<65535)
660  * @param[in] suspend Whether to suspend timer during change.
661  *
662  * @return E_NO_ERROR Success
663  * @return E_TOO_LARGE Failed to find suitable pre-scaler values
664  *
665  * @note Assumes prescale has been set to an appropriate value.
666  * @note Only rudimentary range checking.
667  * @note This function will affect all channels of the timer.
668  * @note This value is write-buffered and updated by MOD synchronisation
669  * unless suspend is true.
670  * @note The counter load value (CNTIN) is cleared
671  */
672  static ErrorCode setPeriodInTicks(uint32_t period, bool suspend=false) {
673 
674  // Check if CPWMS is set (affects period)
675  bool centreAlign = (tmr().SC&FTM_SC_CPWMS_MASK);
676 
677  if (centreAlign) {
678  // Centre-aligned period is 2*MOD value but MOD is
679  // limited to 0x7FFF for sensible PWM operation
680 
681  // Halve with rounding
682  period = (period+1)/2;
683  if (period > 0x7FFFUL) {
684  // Attempt to set too long a period
685  usbdm_assert(false, "Interval is too long");
686  return setErrorCode(E_TOO_LARGE);
687  }
688  }
689  else {
690  // Left-aligned period is MOD+1 value
691  period = period-1;
692  if (period > 0xFFFF) {
693  // Attempt to set too long a period
694  usbdm_assert(false, "Interval is too long");
695  return setErrorCode(E_TOO_LARGE);
696  }
697  }
698  uint32_t sc;
699  if (suspend) {
700  sc = tmr().SC;
701  tmr().SC = 0;
702  }
703  // Start counter from zero
704  tmr().CNTIN = 0;
705 
706  // Change modulo
707  tmr().MOD = period;
708 
709  if (suspend) {
710  // Restart timer
711  tmr().SC = sc;
712  }
713  // OK period
714  return E_NO_ERROR;
715  }

717  /**
718  * Set period
719  *
720  * @param[in] period Period in seconds as a float
721  * @param[in] suspend Whether to suspend timer during change.
722  *
723  * @return E_NO_ERROR => success
724  * @return E_TOO_SMALL => failed to find suitable values
725  * @return E_TOO_LARGE => failed to find suitable values
726  *
727  * @note Adjusts Timer pre-scaler to appropriate value.
728  * @note This function will affect all channels of the timer.
729  * @note The Timer is stopped while being modified
730  * @note The counter load value (CNTIN) is cleared
731  */
732  static ErrorCode setPeriod(float period, bool suspend=false) {
733  float inputClock = Info::getInputClockFrequency();
734  int prescaleFactor=1;
735  int prescalerValue=0;
736 
737  // Maximum period value in ticks
738  uint32_t maxPeriodInTicks = 65536;
739 
740  // Check if CPWMS is set (affects period calculation)
741  if (tmr().SC&FTM_SC_CPWMS_MASK) {
742  // Centre-aligned period is ~double the MOD value but MOD is
743  // limited to 0x7FFF for sensible PWM operation so
744  // period in ticks is limited to 2*0x7FFF
745  maxPeriodInTicks = 65534;
746  }
747  while (prescalerValue<=7) {
748  float clock = inputClock/prescaleFactor;
749  uint32_t periodInTicks = round(period*clock);
750  if (periodInTicks < Info::minimumResolution) {
751  usbdm_assert(false, "Interval is too short");
752  // Too short a period for minimum resolution
753  return setErrorCode(E_TOO_SMALL);
754  }
755  if (periodInTicks <= maxPeriodInTicks) {
756  // Disable timer to change prescaler
757  uint32_t sc = tmr().SC;
758  tmr().SC = 0;
759  (void)tmr().SC;
760  setPeriodInTicks(periodInTicks, suspend);
761  tmr().SC = (sc&~FTM_SC_PS_MASK)|FTM_SC_PS(prescalerValue);
762  return E_NO_ERROR;
763  }
764  prescalerValue++;
765  prescaleFactor <<= 1;
766  }
767  // Too long a period
768  usbdm_assert(false, "Interval is too long");
769  return setErrorCode(E_TOO_LARGE);
770  }
