541  /**
542  * Handle OUT [Receive, device <- host, DATA0/1]
543  */
544  void handleOutToken() {
545  // console.WRITE("Out(),");
546 
547  uint8_t transferredSize = 0;
548 
549  // Toggle DATA0/1 for next transaction
550  fDataToggle = !fDataToggle;
551 
552  switch (fState) {
553  case EPDataOut: // Receiving a sequence of OUT transactions
554  // Save the data from the Receive buffer
555  transferredSize = saveRxData();
556  if ((transferredSize < fEndpointSize) || (fDataRemaining == 0)) {
557  // Completed DATA stage if undersize transaction or received expected number of bytes
558 
559  if (fEndPointType == EndPointType_Control) {
560  // Just completed a series of OUT transfers on Control endpoint -
561  // Do empty status transaction transmission - no response expected
562  setDataToggle(DataToggle_1);
563  startTxStage(EPStatusIn);
564 
565  // Make ready for next SETUP stage
566  setDataToggle(DataToggle_0);
567  startRxStage(EPStatusIn);
568  }
569  else {
570  // Bulk or interrupt
571  // Do callback after DATA OUT stage with no state change
572  fState = fCallback(EPDataOut);
573  }
574  }
575  else {
576  // Set up for next OUT DATA 0/1 transaction
577  startRxTransaction();
578  }
579  break;
580 
581  case EPStatusOut: // Done OUT transaction as a status handshake from host (IN CONTROL transfer)
582  // Make ready for SETUP stage
583  Endpoint::setDataToggle(DataToggle_0);
584  startRxStage(EPIdle);
585 
586  // Execute callback function after STATUS IN transaction
587  fCallback(EPStatusOut);
588  break;
589 
590  // We don't expect an OUT token while in the following states
591  case EPDataIn: // Doing a sequence of IN transactions (until data count <= fEndpointSize)
592  case EPStatusIn: // Just done an IN transaction as a status handshake
593  case EPIdle: // Idle
594  case EPStall: // Not used
595  default:
596  console.WRITE("Unexpected OUT, ep=").WRITE(fEndpointNumber).WRITE(", s=").WRITELN(getStateName());
597  fState = EPIdle;
598  break;
599  }
