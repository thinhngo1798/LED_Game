329  /**
330  * Enter Run Mode.
331  *
332  * This may be used to change between supported RUN modes (RUN, VLPR, HSRUN).\n
333  * Only the following transitions are allowed: VLPR <-> RUN <-> HSRUN.
334  *
335  * @param[in] smcRunMode Mode to enter
336  *
337  * @return E_NO_ERROR No error
338  * @return E_ILLEGAL_POWER_TRANSITION Cannot transition to smcRunMode from current run mode
339  */
340  static ErrorCode enterRunMode(SmcRunMode smcRunMode) {
341 #ifdef SMC_PMPROT_AHSRUN
342  SmcStatus smcStatus = getStatus();
343 #endif
344  switch(smcRunMode) {
345  case SmcRunMode_Normal:
346  smc().PMCTRL = (smc().PMCTRL&~SMC_PMCTRL_RUNM_MASK)|smcRunMode;
347  // Wait for power status to change
348  while (getStatus() != SmcStatus_RUN) {
349  __asm__("nop");
350  }
351  break;
352 #ifdef SMC_PMPROT_AHSRUN
353  case SmcRunMode_HighSpeed:
354  if (smcStatus != SmcStatus_RUN) {
355  // Can only transition from RUN mode
356  return setErrorCode(E_ILLEGAL_POWER_TRANSITION);
357  }
358  smc().PMCTRL = (smc().PMCTRL&~SMC_PMCTRL_RUNM_MASK)|smcRunMode;
359  // Wait for power status to change
360  while (getStatus() != SmcStatus_HSRUN) {
361  __asm__("nop");
362  }
363  break;
364 #endif
365  case SmcRunMode_VeryLowPower:
366 #ifdef SMC_PMPROT_AHSRUN
367  if (smcStatus != SmcStatus_RUN) {
368  // Can only transition from RUN mode
369  return setErrorCode(E_ILLEGAL_POWER_TRANSITION);
370  }
371 #endif
372  smc().PMCTRL = (smc().PMCTRL&~SMC_PMCTRL_RUNM_MASK)|smcRunMode;
373  // Wait for power status to change
374  while (getStatus() != SmcStatus_VLPR) {
375  __asm__("nop");
376  }
377  break;
378  default:
379  return setErrorCode(E_ILLEGAL_PARAM);
380  }
381 #ifdef USBDM_SCG_IS_DEFINED
382  // Update clocks as clock change is automatic with run mode change
383  // Update clocks as clock change is automatic
384  Scg::SystemCoreClockUpdate();
385 #endif
386  return E_NO_ERROR;
387  }

441  /**
442  * Enter Wait Mode (WAIT, VLPW)\n
443  * (ARM core SLEEP mode)
444  *
445  * The processor will stop execution and enter WAIT/VLPW mode.\n
446  * This function can be used to enter normal WAIT mode or VLPW mode
447  * depending upon current run mode.\n
448  * In wait mode the core clock is disabled (no code executing),
449  * but bus clocks are enabled (peripheral modules are operational).
450  *
451  * Possible power mode transitions:
452  * - RUN -> WAIT
453  * - VLPR -> VLPW
454  *
455  * WAIT mode is exited using any enabled interrupt or RESET.
456  *
457  * For Kinetis K:
458  * If in VLPW mode, the statue of the SMC_PMCTRL[LPWUI] bit
459  * determines if the processor exits to VLPR or RUN mode.\n
460  * Use setExitVeryLowPowerOnInterrupt() to modify this action.
461  *
462  * For Kinetis L:
463  * LPWUI does not exist.\n
464  * Exits with an interrupt from VLPW will always be back to VLPR.\n
465  * Exits from an interrupt from WAIT will always be back to RUN.
466  *
467  * @note Some modules include a programmable option to disable them in wait mode.\n
468  * If those modules are programmed to disable in wait mode, they will not be able to
469  * generate interrupts to wake the core.
470  */
471  static void enterWaitMode() {
472  SCB->SCR &= ~SCB_SCR_SLEEPDEEP_Msk;
473  // Make sure write completes
474  (void)(SCB->SCR);
475  __asm volatile( "dsb" ::: "memory" );
476  __asm volatile( "wfi" );
477  __asm volatile( "isb" );
478  }
